#include <iostream>
#include <cmath>

using namespace std;

int fatorial(int x)
{
	if (x==0 || x==1)
		return 1;
	return x*fatorial(x-1);
}

float pi(int x)
{
	float pi;
	for(int i=0; i<x;i++)
		pi += pow(-1,i) * 4/(2*i+1);
	return pi;
}


float euler(int x)
{
	float euler;
	for(int y=0; y<x;y++)
		euler+= 1.0/fatorial(y);
	
	return euler;
}

int main()
{
	int x;
	cin >> x;
	cout << pi(x) << endl;
	return 0;
}
